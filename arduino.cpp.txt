#include <ArduinoBLE.h>
#include <Wire.h>

BLEService temperatureService("0000fff0-0000-1000-8000-00805f9b34fb");
BLECharacteristic temperatureCharacteristic("0000fff0-0000-1000-8000-00805f9b34fb", BLEWrite | BLERead | BLENotify, 20);

void setup() {
  Serial.begin(9600);
  while (!Serial);

  if (!BLE.begin()) {
    Serial.println("starting BLE failed!");
    while (1);
  }

  BLE.setLocalName("BLE");
  BLE.setDeviceName("BLE");

  BLE.addService(temperatureService);
  temperatureService.addCharacteristic(temperatureCharacteristic); // Dodanie charakterystyki do serwisu

  temperatureCharacteristic.writeValue("0.0");
  BLE.advertise();

  Serial.println("Bluetooth device active, waiting for connections...");
}

void loop() {
  BLEDevice central = BLE.central();

  if (central) {
    Serial.print("Connected to central: ");
    Serial.println(central.address());

    while (central.connected()) {
      String message = "Hello from Wemos!";
      temperatureCharacteristic.writeValue(message.c_str());
      Serial.println("Message sent: " + message); // Dodanie zapisu w logach
      delay(1000);
    }

    Serial.print("Disconnected from central: ");
    Serial.println(central.address());
  }
}
